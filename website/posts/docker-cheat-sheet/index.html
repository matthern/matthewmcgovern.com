Docker Cheat Sheet

Some useful docker commands.

Matthew McGovern · May 9, 2023

Install Docker on Rocky/Red Hat 9

# Check for updates
sudo dnf check-update

# Add Docker repository
sudo dnf config-manager --add-repo https://download.docker.com/linux/centos/docker-ce.repo

# Install Docker
sudo dnf install docker-ce docker-ce-cli containerd.io

# Start and enable Docker
sudo systemctl start docker
sudo systemctl enable docker

# Add user to docker group
sudo usermod -aG docker $(whoami)
sudo usermod -aG docker username

Building Docker Images

# Build image from Dockerfile in current directory
docker build .

# Login to Docker Hub
docker login

Tag, Pull, and Push Images

# Tag an image
docker tag reponame/imagename:tagname

# Pull an image
docker pull reponame/imagename:tagname

# Push an image
docker push reponame/imagename:tagname

Clean Up

# Remove dangling images
docker prune

# List dangling images
docker images --filter "dangling=true"

# Remove dangling and unused images
docker prune -a

# Remove image
docker rmi <imagename>

# Remove container
docker rm <container>

Container Management

# Find info in container (grep for yml file, ports, env variables)
docker inspect <container name>

# Show running containers
docker ps 

# Show all containers (including stopped)
docker ps -a

# Show images
docker image ls

Running Containers

# Run a container
docker run <image>
docker container run <image>

# Attach to running container
docker attach <container>

# Run docker-compose with specific file in background
docker-compose -f filename.yml up -d

# Bring down containers
docker-compose -f filename.yml down

# Execute command in container (usually bash)
docker exec -it <container> <command>

Copy Files

# Copy file from container to host
docker cp containername:/usr/share/doc/zabbix-server-pgsql-/create/schema.sql ./schema.sql
